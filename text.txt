  const done = p.tasks.filter((_, i) => checkedTasks[p.id + '_' + i]).length;
  document.getElementById('tasks-done').textContent = done + '/' + p.tasks.length + ' done';
  document.getElementById('tasks-done').style.color = p.color;

  p.tasks.forEach((task, i) => {
    const key = p.id + '_' + i;
    const checked = !!checkedTasks[key];
    const item = document.createElement('div');
    item.className = 'task-item' + (checked ? ' checked' : '');
    item.style.background = checked ? p.color + '0e' : '#0a0a0a';
    item.style.borderColor = checked ? p.color + '33' : '#141414';
    item.innerHTML = `
      <div class="task-checkbox" style="border-color:${p.color};background:${checked ? p.color : 'transparent'}">
        <span class="task-check-icon">✓</span>
      </div>
      <span class="task-text">${task}</span>`;
    item.onclick = () => {
      checkedTasks[key] = !checkedTasks[key];
      saveChecked();
      renderToday();
    };
    list.appendChild(item);
  });

  const tipBox = document.getElementById('tip-box');
  tipBox.style.background = p.color + '0a';
  tipBox.style.border = '1px solid ' + p.color + '22';
  document.getElementById('tip-text').textContent = p.tip;
}

// ---- TABS ----
function switchTab(tab) {
  currentTab = tab;
  document.querySelectorAll('.tab-btn').forEach(b => b.classList.toggle('active', b.dataset.tab === tab));
  document.querySelectorAll('.tab-panel').forEach(p => p.classList.toggle('active', p.id === 'tab-' + tab));

  const aiPanel = document.getElementById('ai-panel');
  const content = document.getElementById('content');

  if (tab === 'ai') {
    content.style.display = 'none';
    aiPanel.style.display = 'flex';
    aiPanel.style.flex = '1';
    aiPanel.style.minHeight = '0';
    updateAIContext();
    renderQuickPrompts();
  } else {
    content.style.display = 'block';
    aiPanel.style.display = 'none';
    if (tab === 'today') renderToday();
    if (tab === 'phases') renderPhaseList();
  }
}

document.querySelectorAll('.tab-btn').forEach(btn => {
  btn.onclick = () => switchTab(btn.dataset.tab);
});

// ---- DAY CONTROLS ----
function setDay(d) {
  currentDay = Math.max(1, Math.min(180, d));
  updateHeader();
  if (currentTab === 'today') renderToday();
  if (currentTab === 'phases') renderPhaseList();
  if (currentTab === 'ai') updateAIContext();
}

document.getElementById('btn-minus').onclick = () => setDay(currentDay - 1);
document.getElementById('btn-plus').onclick = () => setDay(currentDay + 1);
document.getElementById('day-slider').oninput = e => setDay(parseInt(e.target.value));

// ---- PLANT DATE ----
function applyPlantDate(dateStr) {
  if (!dateStr) return;
  const planted = new Date(dateStr);
  const diff = Math.floor((Date.now() - planted.getTime()) / 86400000) + 1;
  setDay(Math.max(1, Math.min(diff, 180)));
}

document.getElementById('plant-date').onchange = e => {
  plantDate = e.target.value;
  localStorage.setItem('kk_plant_date', plantDate);
  document.getElementById('reset-date').style.display = plantDate ? 'block' : 'none';
  applyPlantDate(plantDate);
};

document.getElementById('reset-date').onclick = () => {
  plantDate = null;
  localStorage.removeItem('kk_plant_date');
  document.getElementById('plant-date').value = '';
  document.getElementById('reset-date').style.display = 'none';
  setDay(1);
};

// Restore plant date on load
if (plantDate) {
  document.getElementById('plant-date').value = plantDate;
  document.getElementById('reset-date').style.display = 'block';
  applyPlantDate(plantDate);
}

function addMessage(role, text) {
  const messages = document.getElementById('chat-messages');
  const div = document.createElement('div');

  messages.appendChild(div);

}

function setLoading(val) {

}
